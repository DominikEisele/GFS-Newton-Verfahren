\section{Ziel des Newton Verfahrens}
	\begin{itemize}
		\item Annäherung an die Nullstellen, über eine Linearisierung der Funktion an geeigneten Stellen
		\item Lösung von allen nichtlinearen Gleichungen und Gleichungssystemen
		\item Lösung von Gleichungen in der Form: $f(x) = 0$
	\end{itemize}
	
\section{Vorgehen bei Konstruktion}
	\begin{enumerate}
		\item die Tangente der Funktion $f(x)$ im Startpunkt bestimmen
		\item der Schnittpunkt der Tangente mit der x-Achse ist der neue Startpunkt
		\item dieses Verfahren so oft wiederholen bis man die gewünschte Genauigkeit erreicht hat
	\end{enumerate}

\section{Iterationsformel}
	\begin{equation*}
		\boxed{x_{n+1} = x_{n} - \frac{f(x_{n})}{f^\prime(x_{n})}}
	\end{equation*}

	\begin{enumerate}
		\item Startwert in der Nähe der Nullstelle wählen
		\item Iteration so lange durchführen bis das gewünschte Ergebnis erzielt ist
	\end{enumerate}
